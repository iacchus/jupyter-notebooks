def returnScale(tonic='c', mode='major', ownKey = False): modes = [ 'ionian','dorian','phrygian', 'lydian','mixolydian','aeolian', 'locrian','major','minor'] if mode in modes: if mode == 'ionian': mode = 'major' elif mode == 'aeolian': mode = 'minor' else: mymode = 'major' myfunc = eval("scale.{0}Scale".format(mode.capitalize())) pitches = myfunc(tonic).getPitches() scaleSharps = key.pitchToSharps(tonic, mode) myKeySig = key.KeySignature(scaleSharps) mystream = stream.Stream() if ownKey: mystream.append(myKeySig) for x in range(0, len(pitches)): mynote = note.Note(pitches[x]) mynote.addLyric(interval.notesToInterval(pitches[0], mynote).directedName) mynote.addLyric(interval.notesToInterval(pitches[(x-1 if x>0 else 0)], mynote).directedName) mystream.append(mynote) return mystream
